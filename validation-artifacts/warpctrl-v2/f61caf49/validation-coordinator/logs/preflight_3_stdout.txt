'--output-format=[Set the output format]:OUTPUT_FORMAT:((json\:"Output as JSON"
ndjson\:"Output as newline-delimited JSON"
pretty\:"Output as human-readable text"
text\:"Output as plain text"))' \
'-h[Print help (see more with '\''--help'\'')]' \
'--help[Print help (see more with '\''--help'\'')]' \
'-V[Print version]' \
'--version[Print version]' \

"*::: :->warpctrl" \
&& ret=0

    (warpctrl)
        words=($line[1] "${words[@]}")

        curcontext="${curcontext%:*:*}:warpctrl-command-$line[1]:"

            (instance)
_arguments "${_arguments_options[@]}" : \
'--output-format=[Set the output format]:OUTPUT_FORMAT:((json\:"Output as JSON"
ndjson\:"Output as newline-delimited JSON"

pretty\:"Output as human-readable text"
text\:"Output as plain text"))' \
